Example request:

    --get "http://localhost:7700/api/v1/posts?page=1&per_page=10" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"page\": 19,
    \"per_page\": 4
}"

const url = new URL(
    "http://localhost:7700/api/v1/posts"
);

const params = {
    "page": "1",
    "per_page": "10",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "page": 19,
    "per_page": 4
};
